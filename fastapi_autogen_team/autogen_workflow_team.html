 97def handle_tool_responses(
 98    self,
 99    message: Dict,
100    sender: Agent,
101    queue: Queue,
102    index: int,
103    iostream: IOStream,
104    streaming_message: str,
105    *args,
106    **kwargs,
107) -> str:
108    """Handles messages containing tool responses, including printing and queuing."""
109    if message.get("role") == "tool":
110        queue.put(
111            {
112                "index": index,
113                "delta": {"role": "assistant", "content": streaming_message},
114                "finish_reason": "stop",
115            }
116        )
117    for tool_response in message["tool_responses"]:
118        index += 1
119        self._print_received_message(
120            message=tool_response,
121            sender=sender,
122            queue=queue,
123            index=index,
124            *args,
125            **kwargs,
126        )
127    if message.get("role") == "tool":

129    return streaming_message

132def handle_function_tool_message(message: Dict, iostream: IOStream, streaming_message: str) -> str:
133    """Handles messages from function or tool calls."""
134    id_key = "name" if message["role"] == "function" else "tool_call_id"
135    id_str = message.get(id_key, "No id found")
136    func_print = f"***** Response from calling {message['role']} ({id_str}) *****"
137
138    iostream.print(colored(func_print, "green"), flush=True)
139    streaming_message += f"{func_print}\n"
140
141    iostream.print(message["content"], flush=True)
142    streaming_message += f"{message['content']}\n"
143
144    iostream.print(colored("*" * len(func_print), "green"), flush=True)
145    streaming_message += f"{'*' * len(func_print)}\n"
146    return streaming_message

149def handle_regular_message(self, message: Dict, iostream: IOStream, streaming_message: str) -> str:
150    """Handles regular messages (not tool or function calls)."""
151    content = message.get("content")
152    if content is not None:
153        if "context" in message:
154            content = OpenAIWrapper.instantiate(
155                content,
156                message["context"],
157                self.llm_config and self.llm_config.get("allow_format_str_template", False),
158            )
159        iostream.print(content_str(content), flush=True)
160        streaming_message += f"{content_str(content)}\n"
161
162    if "function_call" in message and message["function_call"]:
163        streaming_message = handle_suggested_function_call(message["function_call"], iostream, streaming_message)
164
165    if "tool_calls" in message and message["tool_calls"]:
166        streaming_message = handle_suggested_tool_calls(message["tool_calls"], iostream, streaming_message)
167    return streaming_message

368    def run(self, message: str, stream: bool = False) -> ChatResult:
369        """Initiates the Autogen workflow and returns the chat history."""
370        if stream:
371            index_counter = {"index": 0}
372            queue = self.queue
373
374            def streamed_print_received_message_with_queue_and_index(self, *args, **kwargs):
375                streamed_print_received_message_with_queue = partial(
376                    streamed_print_received_message, queue=queue, index=index_counter["index"]
377                )
378                bound_method = types.MethodType(streamed_print_received_message_with_queue, self)
379                result = bound_method(*args, **kwargs)
380                index_counter["index"] += 1
381                return result
382
383            self.group_chat_manager_with_intros._print_received_message = types.MethodType(
384                streamed_print_received_message_with_queue_and_index, self.group_chat_manager_with_intros
385            )
386
387        try:
388            chat_history = self.user_proxy.initiate_chat(self.group_chat_manager_with_intros, message=message)
389
390            if stream and self.queue is not None:
391                self.queue.put("[DONE]")
392
393            return chat_history
394
395        except Exception as e:
396            # Handle any other exceptions
397            error_message = {"error": "Workflow Error", "details": str(e), "type": "system_error"}
398
399            logger.error(f"Workflow error: {str(e)}")
400
401            if stream and self.queue is not None:
402                self.queue.put(
403                    {
404                        "index": index_counter["index"] if "index_counter" in locals() else 0,
405                        "delta": {"role": "assistant", "content": f"System error occurred: {str(e)}"},
406                        "finish_reason": "error",
407                    }
408                )
409                self.queue.put("[DONE]")
410
411            # Return a chat result with error information
412            return ChatResult(
413                chat_history=[{"role": "error", "content": f"System error occurred: {str(e)}", "error": error_message}],
414                summary="Conversation failed due to system error",
415            )
